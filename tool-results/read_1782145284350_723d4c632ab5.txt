   258→    async function init() {
   259→      // Set up auth subscription
   260→      subscription = supabase.auth.onAuthStateChange(
   261→        async (_event, session) => {
   262→          if (!mounted) return;
   263→
   264→          // ── Null session: NEVER clear state from this handler ──
   265→          // signOut() fires SIGNED_OUT asynchronously (queued by GoTrue).
   266→          // A delayed SIGNED_OUT from a previous signOut can arrive AFTER
   267→          // a new signInWithPassword() + setStatus('authenticated'), wiping

   272→          // exists. Null-session events are handled by:
   273→          // - logout() — clears state directly before calling signOut.
   274→          // - 10s heartbeat — detects session eviction server-side.
   275→          // - GoTrue's auto-refresh — handles token renewal internally.
   276→          if (!session) return;

   278→          // ── Not authenticated: LoginScreen handles sign-in directly ──

   280→          // login handler sets auth state directly via setAuth + setStatus.
   281→          if (useAuthStore.getState().status !== 'authenticated') return;
   282→

   284→          if (!navigator.onLine) return;

   287→            // Session refreshed or user updated — re-sync auth + profile
   288→            setAuth(session.user, session);
   289→            const profile = await getMyProfile();
   290→            if (profile) setProfile(profile);
   291→          } catch {
   292→            // Non-critical: transient error (network blip, Supabase hiccup).
   293→            // The next heartbeat or online event will retry profile sync.
   294→          }
   295→        },
   296→      ).data.subscription;
   297→
   298→      // Validate session
   299→      try {
   300→        const { data: { session } } = await supabase.auth.getSession();
   301→
   302→        if (!session) {
   303→          // OFFLINE RECOVERY: getSession() may return null when Supabase cleared
   304→          // the stored session during a failed token refresh. If we're offline
   305→          // and have a cached session in localStorage, recover from it instead
   306→          // of wiping the auth state.
   307→          if (!navigator.onLine) {
   308→            try {
   309→              const raw = localStorage.getItem('streamvault-auth-token');
   310→              if (raw) {
   311→                const cached = JSON.parse(raw);
   312→                if (cached?.access_token) {
   313→                  // Keep the authenticated state from bootstrapFromCache
   314→                  return;
   315→                }
   316→              }
   317→            } catch { /* ignore parse errors */ }
   318→          }
   319→          if (mounted) {
   320→            setProfile(null);
   321→            setStatus('unauthenticated');
   322→          }
   323→          return;
   324→        }
   325→
   326→        if (mounted) setAuth(session.user, session);
   327→
   328→        const metaCompleted = !!session.user.user_metadata?.profile_completed;
   329→        const profile = await getMyProfile();
   330→        if (!mounted) return;
   331→
   332→        setProfile(profile);
   333→
   334→        if (metaCompleted || (profile && profile.display_name.trim().length > 0)) {
   335→          setStatus('authenticated');
   336→          if (session.access_token) {
   337→            fetchWatchlistFromServer(session.access_token);
   338→            // Bump updated_at to signal user activity (throttled to 1h)
   339→            touchProfile(profile?.updated_at);
   340→          }
   341→        } else {
   342→          setStatus('needs_profile');
   343→        }
   344→      } catch {
   345→        if (mounted) {
   346→          const current = useAuthStore.getState().status;
   347→          // OFFLINE RECOVERY: bootstrapFromCache() already set status to
   348→          // 'authenticated'. Don't overwrite it with 'unauthenticated' just
   349→          // because getMyProfile() failed (expected when offline).
   350→          if (current === 'loading') {
   351→            const cached = getCachedProfile();
   352→            if (cached && cached.display_name.trim().length > 0) {
   353→              setProfile(cached);
   354→              setStatus('authenticated');
   355→              // Only mark offline if actually offline — getMyProfile() can
   356→              // fail for non-offline reasons (RLS, Supabase hiccup, etc.)
   357→              if (!navigator.onLine) setOffline(true);
   358→            } else {
   359→              setStatus('unauthenticated');
   360→            }
   361→          } else if (current === 'authenticated') {
   362→            // Already recovered by bootstrap — only mark offline if actually offline
   363→            if (!navigator.onLine) setOffline(true);
   364→          }
   365→        }
   366→      }
   367→    }
   368→
   369→    init();
   370→
   371→    return () => {
   372→      mounted = false;
   373→      if (subscription) subscription.unsubscribe();
   374→    };
