    74→// ── REST fallback detection ────────────────────────────────
    75→// When the Supabase WebSocket silently dies, channel.send() falls back to
    76→// REST without throwing. We detect this by checking if send() returns a
    77→// thenable (Promise). If we see 3+ REST fallbacks in a 10s window, the
    78→// WebSocket is confirmed dead and we recreate the channel.
    79→let _restFallbackCount = 0
    80→let _restFallbackResetTimer: ReturnType<typeof setTimeout> | null = null
    81→const REST_FALLBACK_THRESHOLD = 3 // consecutive REST sends before recreating channel
    82→const REST_FALLBACK_WINDOW_MS = 10_000 // time window for counting consecutive REST sends
    83→
    84→function detectRestFallback(channel: RealtimeChannel, result: unknown): void {
    85→  // channel.send() returns a Promise when it falls back to REST.
    86→  // When using WebSocket, it returns undefined or a non-thenable.
    87→  if (result && typeof result === 'object' && 'then' in result) {
    88→    _restFallbackCount++
    89→    if (_restFallbackResetTimer) clearTimeout(_restFallbackResetTimer)
    90→    _restFallbackResetTimer = setTimeout(() => {
    91→      _restFallbackCount = 0
    92→    }, REST_FALLBACK_WINDOW_MS)
    93→
    94→    if (_restFallbackCount >= REST_FALLBACK_THRESHOLD) {
    95→      _restFallbackCount = 0
    96→      if (_restFallbackResetTimer) clearTimeout(_restFallbackResetTimer)
    97→      // WebSocket is dead — recreate the channel to restore WS connection
    98→      if (channel === _partyChannel) {
    99→        console.warn('[WatchParty] Detected sustained REST fallback — recreating party channel')
   100→        recreatePartyChannel()
   101→      } else if (channel === _invitesChannel) {
   102→        console.warn('[WatchParty] Detected sustained REST fallback — recreating invites channel')
   103→        recreateInvitesChannel()
   104→      }
   105→    }
   106→  } else {
   107→    // send() used WebSocket — reset counter
   108→    _restFallbackCount = 0
   109→    if (_restFallbackResetTimer) clearTimeout(_restFallbackResetTimer)
   110→  }
   111→}

   122→// ── Broadcast helper with error handling ────────────────────
   123→// Centralizes all Supabase Realtime broadcast sends with:
   124→//   - Always tries WebSocket first (fast, real-time)
   125→//   - Falls back to httpSend() for that specific message when WebSocket is down
   126→//   - Schedules channel resubscribe to recover WebSocket connection
   127→//
   128→// IMPORTANT: We do NOT track a global "WebSocket connected" flag.
   129→// The old approach had a shared _wsConnected flag that, once set to false
   130→// by ANY channel's disconnect, permanently disabled WebSocket sends for
   131→// ALL channels. This caused a cascade failure: one channel's WebSocket
   132→// drop → all channels fall back to REST → WebRTC signaling becomes
   133→// unreliable → ICE negotiation fails → no voice + no PTT indicators.
   134→
   135→// Track channels that are pending resubscription to avoid duplicate attempts
   136→const _resubscribePending = new WeakSet<RealtimeChannel>()
   137→
   138→function wpBroadcast(
   139→  channel: RealtimeChannel | null,
   140→  payload: WpBroadcastEvent,
   141→  label: string,
   142→): void {
   143→  if (!channel) {
   144→    console.warn(`[WatchParty] Broadcast skipped (${label}): no channel`)
   145→    return
   146→  }
   147→
   148→  try {
   149→    const result = channel.send({
   150→      type: 'broadcast',
   151→      event: 'wp',
   152→      payload,
   153→    })
   154→    // Detect if send() fell back to REST (returns a Promise when REST,
   155→    // undefined/thenable when WebSocket). If sustained REST detected,
   156→    // recreate the channel to restore WebSocket connectivity.
   157→    detectRestFallback(channel, result)
   158→    return
   159→  } catch {
   160→    // send() threw — WebSocket is likely down, try httpSend as fallback
   161→    try {
   162→      channel.httpSend('wp', payload).then((res: { success: boolean; error?: string }) => {
   163→        if (res.success === false) {
   164→          console.error(`[WatchParty] httpSend failed (${label}):`, res.error)
   165→        }
   166→      }).catch((err: Error) => {
   167→        console.error(`[WatchParty] httpSend error (${label}):`, err.message)
   168→      })
   169→      // Schedule resubscribe since send() threw (WebSocket definitely down)
   170→      scheduleChannelResubscribe(channel)
   171→    } catch {
   172→      // Completely failed — nothing more we can do
   173→    }
   174→  }
   175→}
   176→
   177→/** Schedule a channel resubscribe to recover the WebSocket connection. */
   178→function scheduleChannelResubscribe(channel: RealtimeChannel): void {
   179→  if (_resubscribePending.has(channel)) return
   180→  _resubscribePending.add(channel)
   181→
   182→  // Wait 2 seconds before resubscribing to avoid hammering the server
   183→  // if multiple broadcasts fail in quick succession (e.g., ICE candidates)
   184→  setTimeout(() => {
   185→    _resubscribePending.delete(channel)
   186→    const state = channel.state
   187→    // Supabase Realtime uses 'joined' for the subscribed state.
   188→    // Valid states: 'closed', 'closing', 'joined', 'joining', 'leaving', 'timed_out'
   189→    if (state === 'joined') return
   190→
   191→    // CRITICAL: channel.subscribe() can only be called ONCE per channel
   192→    // instance. If the channel state is 'closed' or 'timed_out', the
   193→    // underlying WebSocket is dead and subscribe() will throw
   194→    // 'tried to join multiple times'. The only fix is to create a
   195→    // brand-new channel instance.
   196→    if (channel === _partyChannel) {
   197→      console.log('[WatchParty] Party channel dead (state:', state, ') — recreating channel')
   198→      recreatePartyChannel()
   199→    } else if (channel === _invitesChannel) {
   200→      console.log('[WatchParty] Invites channel dead (state:', state, ') — recreating channel')
   201→      recreateInvitesChannel()
   202→    }
   203→  }, 2000)
   204→}

   206→/**
   207→ * Recreate the party channel when the WebSocket dies.
   208→ * Supabase Realtime's subscribe() can only be called once per channel instance.
   209→ * When the channel reaches 'closed' or 'timed_out', the only option is
   210→ * to create a fresh channel. This preserves the voice manager and peer
   211→ * connections — only the signaling transport is replaced.
   212→ */
   213→function recreatePartyChannel(): void {
   214→  if (!_currentPartyId) return
   215→  // Cooldown: don't recreate more than once every 10 seconds.
   216→  // Cascading recreations (REST fallback → recreate → fail → recreate)
   217→  // were causing a 429 rate limit storm on the API.
   218→  const now = Date.now()
   219→  if (now - _lastChannelRecreation < 10_000) return
   220→  _lastChannelRecreation = now
   221→
   222→  const partyId = _currentPartyId
   223→  const userId = useAuthStore.getState().user?.id
   224→  if (!userId) return
   225→  const profile = useAuthStore.getState().profile
   226→
   227→  // Preserve the voice manager across channel recreation.
   228→  // unsubscribePartyChannel destroys the voice manager, but during
   229→  // a recreation we only need a new signaling transport — the
   230→  // existing WebRTC peer connections should survive. Destroying
   231→  // them mid-PTT causes NPE crashes and permanent voice loss.
   232→  const savedVoiceManager = _voiceManager
   233→  const savedVoiceReady = _voiceReady
   234→
   235→  // Remove dead channel so subscribePartyChannel doesn't early-return.
   236→  _partyChannel = null
   237→  // Temporarily null the voice manager so unsubscribePartyChannel skips it
   238→  _voiceManager = null
   239→  _voiceReady = false
   240→
   241→  subscribePartyChannel(partyId, userId, profile?.display_name || 'Unknown', profile?.avatar_url || null)
   242→
   243→  // subscribePartyChannel creates a new voice manager because _voiceManager
   244→  // was null. But we want to KEEP the existing one (with its established
   245→  // peer connections). Restore the saved manager and destroy the new one.
   246→  if (savedVoiceManager) {
   247→    if (_voiceManager && _voiceManager !== savedVoiceManager) {
   248→      // A new manager was created — destroy it and restore the saved one
   249→      _voiceManager.destroy()
   250→    }
   251→    _voiceManager = savedVoiceManager
   252→    _voiceReady = savedVoiceReady
   253→    // Update signal sender to use the new channel
   254→    if (_partyChannel) {
   255→      _voiceManager.setSignalSender((targetUserId, signal) => {
   256→        if (!_partyChannel) return
   257→        let payload: WpWebrtcEvent
   258→        switch (signal.type) {
   259→          case 'offer':
   260→            payload = { t: 'webrtc-offer', targetUserId, fromUserId: userId, sdp: signal.sdp! }
   261→            break
   262→          case 'answer':
   263→            payload = { t: 'webrtc-answer', targetUserId, fromUserId: userId, sdp: signal.sdp! }
   264→            break
   265→          case 'ice-candidate':
   266→            payload = { t: 'webrtc-ice', targetUserId, fromUserId: userId, candidate: signal.candidate! }
   267→            break
   268→        }
   269→        wpBroadcast(_partyChannel!, payload, `webrtc-${payload.t}`)
   270→      })
   271→    }
   272→    console.log('[WatchParty] Preserved voice manager across channel recreation')
   273→  }
   274→}

   412→async function wpApi(
   413→  action: string,
   414→  payload: Record<string, unknown> = {},
   415→): Promise<{ ok: boolean; data: Record<string, unknown>; error?: string }> {
   416→  const token = await getAuthToken()
   417→  if (!token) {
   418→    return { ok: false, data: {}, error: 'Not authenticated' }
   419→  }
   420→
   421→  try {
   422→    const res = await fetch('/api/watch-party', {
   423→      method: 'POST',
   424→      headers: {
   425→        'Content-Type': 'application/json',
   426→        Authorization: `Bearer ${token}`,
   427→      },
   428→      body: JSON.stringify({ action, ...payload }),
   429→    })
   430→
   431→    const json = await res.json()
   432→
   433→    if (!res.ok) {
   434→      console.error(`[WatchParty] API error (${res.status}):`, json.error || 'Unknown error')
   435→      return { ok: false, data: {}, error: json.error || `Request failed (${res.status})` }
   436→    }
   437→
   438→    return { ok: true, data: json }
   439→  } catch (err) {
   440→    const message = err instanceof Error ? err.message : 'Network error'
   441→    console.error('[WatchParty] API network error:', message)
   442→    return { ok: false, data: {}, error: message }
   443→  }
   444→}

   446→// ── Sync debounce — prevents DB write flood ────────────────

   461→function debouncedSyncToDb(partyId: string, currentTime: number, isPlaying: boolean): void {
   462→  const now = Date.now()
   463→  if (now - _lastSyncDbWrite < SYNC_DB_DEBOUNCE_MS) return
   464→  _lastSyncDbWrite = now
   465→  wpApi('sync', { partyId, currentTime, isPlaying }).catch(() => { /* non-critical */ })
   466→}

   470→async function fetchPartyFromDb(partyId: string): Promise<WatchPartyData | null> {
   471→  try {
   472→    const { data: party, error } = await supabase
   473→      .from('watch_parties')
   474→      .select('*')
   475→      .eq('id', partyId)
   476→      .maybeSingle()
   477→
   478→    if (error || !party) return null
   479→
   480→    const { data: memberRows } = await supabase
   481→      .from('watch_party_members')
   482→      .select('user_id, status')
   483→      .eq('party_id', partyId)
   484→      .in('status', ['joined', 'invited'])
   485→
   486→    const memberUserIds = (memberRows || []).map((m) => m.user_id)
   487→    const memberStatusMap = new Map((memberRows || []).map((m) => [m.user_id, m.status]))
   488→
   489→    const { data: profiles } = await supabase
   490→      .from('profiles')
   491→      .select('id, display_name, avatar_url')
   492→      .in('id', memberUserIds.length > 0 ? memberUserIds : ['__none__'])
   493→
   494→    const profileMap = new Map((profiles || []).map((p) => [p.id, p]))
   495→
   496→    const partyMembers: PartyMember[] = memberUserIds.map((uid) => {
   497→      const profile = profileMap.get(uid)
   498→      return {
   499→        userId: uid,
   500→        displayName: profile?.display_name || 'Unknown',
   501→        avatarUrl: profile?.avatar_url || null,
   502→        isHost: uid === party.host_id,
   503→        isTalking: false,
   504→        memberStatus: (memberStatusMap.get(uid) as 'joined' | 'invited') || 'joined' as const,
   505→      }
   506→    })
   507→
   508→    return {
   509→      id: party.id,
   510→      hostId: party.host_id,
   511→      contentId: party.content_id,
   512→      mediaType: party.media_type as 'movie' | 'tv' | null,
   513→      season: party.season,
   514→      episode: party.episode,
   515→      contentTitle: party.content_title,
   516→      contentPoster: party.content_poster,
   517→      status: party.status as 'waiting' | 'playing' | 'ended',
   518→      members: partyMembers,
   519→      playbackState: {
   520→        isPlaying: party.is_playing,
   521→        currentTime: party.playback_time || 0,
   522→        duration: 0,
   523→        pausedBy: party.paused_by,
   524→      },
   525→      createdAt: new Date(party.created_at).getTime(),
   526→    }
   527→  } catch {
   528→    return null
   529→  }
   530→}

   532→async function fetchPendingInvites(userId: string): Promise<WatchPartyInvitation[]> {
   533→  try {
   534→    const { data: memberRows } = await supabase
   535→      .from('watch_party_members')
   536→      .select('party_id')
   537→      .eq('user_id', userId)
   538→      .eq('status', 'invited')
   539→
   540→    if (!memberRows || memberRows.length === 0) return []
   541→
   542→    const invitations: WatchPartyInvitation[] = []
   543→    for (const row of memberRows) {
   544→      const party = await fetchPartyFromDb(row.party_id)
   545→      if (!party || party.status === 'ended') continue
   546→
   547→      const hostMember = party.members.find((m) => m.isHost)
   548→      invitations.push({
   549→        partyId: row.party_id,
   550→        hostId: party.hostId,
   551→        hostName: hostMember?.displayName || 'Unknown',
   552→        hostAvatarUrl: hostMember?.avatarUrl || null,
   553→        memberCount: party.members.length,
   554→        receivedAt: Date.now(),
   555→      })
   556→    }
   557→    return invitations
   558→  } catch {
   559→    return []
   560→  }
   561→}

   563→async function checkActiveParty(userId: string) {
   564→  try {
   565→    // Guard: if already in a party with an active channel, don't overwrite state.
   566→    // This prevents auth token refreshes from destroying the current party session.
   567→    const currentStore = useWatchPartyStore.getState()
   568→    if (currentStore.isInParty && _partyChannel && _currentPartyId) {
   569→      return
   570→    }
   571→
   572→    const { data: memberRows } = await supabase
   573→      .from('watch_party_members')
   574→      .select('party_id')
   575→      .eq('user_id', userId)
   576→      .eq('status', 'joined')
   577→      .order('joined_at', { ascending: false })
   578→      .limit(1)
   579→
   580→    if (!memberRows || memberRows.length === 0) return
   581→
   582→    const partyId = memberRows[0].party_id
   583→    const partyData = await fetchPartyFromDb(partyId)
   584→
   585→    if (!partyData || partyData.status === 'ended') {
   586→      // Clean up stale membership via API
   587→      await wpApi('leave', { partyId })
   588→      return
   589→    }
   590→
   591→    const store = useWatchPartyStore.getState()
   592→    store.setCurrentParty(partyData)
   593→    useWatchPartyStore.setState({
   594→      isHost: partyData.hostId === userId,
   595→      isRoomVisible: true,
   596→      partyStartTime: partyData.playbackState.currentTime,
   597→    })
   598→
   599→    const profile = useAuthStore.getState().profile
   600→    subscribePartyChannel(partyId, userId, profile?.display_name || 'Unknown', profile?.avatar_url || null)
   601→  } catch {
   602→    // Silent — don't block app load
   603→  }
   604→}

   606→// ── Channel management ─────────────────────────────────────

   608→function subscribeInvitesChannel(userId: string) {
   609→  if (_invitesChannel && _subscribedUserId === userId) return
   610→
   611→  unsubscribeInvitesChannel()
   612→
   613→  _subscribedUserId = userId
   614→
   615→  try {
   616→    // Per-user invite channel — each user only receives their own invites.
   617→    // Previous design used a shared 'wp-invites' channel where ALL users
   618→    // received ALL invite payloads (privacy leak). Now each user subscribes
   619→    // to their own channel, so they only see invites targeting them.
   620→    _invitesChannel = supabase.channel(`wp-invites-${userId}`, {
   621→      config: { broadcast: { self: false } },
   622→    })
   623→
   624→    _invitesChannel.on('broadcast', { event: 'wp' }, (payload: { payload: WpBroadcastEvent }) => {
   625→      const event = payload.payload
   626→      const store = useWatchPartyStore.getState()
   627→
   628→      if (event.t === 'invite') {
   629→        // Double-check targetUserId matches (defense in depth)
   630→        if (event.targetUserId === userId) {
   631→          store.addInvitation({
   632→            partyId: event.partyId,
   633→            hostId: event.hostId,
   634→            hostName: event.hostName,
   635→            hostAvatarUrl: event.hostAvatarUrl,
   636→            memberCount: event.memberCount,
   637→            receivedAt: Date.now(),
   638→          })
   639→        }
   640→      }
   641→
   642→      if (event.t === 'invite-rejected') {
   643→        if (event.targetUserId === userId) {
   644→          toast.info(`${event.displayName} declined your watch party invitation`)
   645→          store.removeInvitation(event.partyId)
   646→        }
   647→      }
   648→    })
   649→
   650→    _invitesChannel.subscribe((status) => {
   651→      if (status === 'SUBSCRIBED') {
   652→        console.log('[WatchParty] Invites channel subscribed successfully')
   653→        _restFallbackCount = 0
   654→      } else if (status === 'CHANNEL_ERROR' || status === 'TIMED_OUT' || status === 'CLOSE') {
   655→        console.error('[WatchParty] Invites channel disconnected:', status)
   656→        // Retry after 5 seconds — must recreate the channel since
   657→        // subscribe() can only be called once per instance
   658→        setTimeout(() => {
   659→          if (_invitesChannel) {
   660→            console.log('[WatchParty] Invites channel dead — recreating')
   661→            recreateInvitesChannel()
   662→          }
   663→        }, 5000)
   664→      }
   665→    })
   666→  } catch (err) {
   667→    console.error('[WatchParty] Invites channel creation failed:', err instanceof Error ? err.message : err)
   668→    _invitesChannel = null
   669→    _subscribedUserId = null
   670→  }
   671→}
   672→
   673→function unsubscribeInvitesChannel() {
   674→  if (_invitesChannel) {
   675→    try { supabase.removeChannel(_invitesChannel) } catch { /* already removed */ }
   676→    _invitesChannel = null
   677→  }
   678→  _subscribedUserId = null
   679→}
   680→
   681→function subscribePartyChannel(
   682→  partyId: string,
   683→  userId: string,
   684→  displayName: string,
   685→  avatarUrl: string | null,
   686→  onSubscribed?: () => void,
   687→) {
   688→  if (_partyChannel && _currentPartyId === partyId) return
   689→
   690→  unsubscribePartyChannel()
   691→
   692→  _currentPartyId = partyId
   693→
   694→  try {
   695→    _partyChannel = supabase.channel(`wp-party-${partyId}`, {
   696→      config: {
   697→        broadcast: { self: false },
   698→        presence: { key: userId },
   699→      },
   700→    })
   701→
   702→    // ── Start content-heal interval IMMEDIATELY (not inside subscribe) ──
   703→    // CRITICAL: Previously this interval was only created inside the
   704→    // subscribe() callback when status === 'SUBSCRIBED'. If the WebSocket
   705→    // never connects (common on PWA/mobile/unstable networks), the
   706→    // interval never starts, and members are stuck with stale data:
   707→    // no content, no status updates, no auto-play. This is the root
   708→    // cause of "content doesn't appear for members" and "start doesn't
   709→    // work for members". Starting it here ensures DB polling runs
   710→    // regardless of WebSocket state.
   711→    if (_contentHealInterval) { clearInterval(_contentHealInterval); _contentHealInterval = null }
   712→    _contentHealInterval = setInterval(async () => {
   713→      if (!_currentPartyId || _syncInProgress) return
   714→      const store = useWatchPartyStore.getState()
   715→      if (!store.currentParty) return
   716→
   717→      // Fast path: detect party ended (host closed app).
   718→      try {
   719→        const { data: statusRow } = await supabase
   720→          .from('watch_parties')
   721→          .select('status')
   722→          .eq('id', _currentPartyId)
   723→          .maybeSingle()
   724→        if (statusRow?.status === 'ended') {
   725→          toast.info('Watch party has ended')
   726→          store.leaveParty()
   727→          unsubscribePartyChannel()
   728→          return
   729→        }
   730→      } catch { /* non-critical */ }
   731→
   732→      // Check content + status from DB.
   733→      _syncInProgress = true
   734→      try {
   735→        const { data: contentRow, error: contentError } = await supabase
   736→          .from('watch_parties')
   737→          .select('content_id, media_type, season, episode, content_title, content_poster, is_playing, playback_time, status')
   738→          .eq('id', _currentPartyId)
   739→          .maybeSingle()
   740→
   741→        if (contentError || !contentRow) return
   742→        const current = useWatchPartyStore.getState()
   743→        if (!current.currentParty) return
   744→
   745→        // Sync content if DB has content that differs from local
   746→        if (contentRow.content_id && contentRow.content_id !== current.currentParty.contentId) {
   747→          current.setPartyContent({
   748→            contentId: contentRow.content_id,
   749→            mediaType: contentRow.media_type as 'movie' | 'tv',
   750→            season: contentRow.season,
   751→            episode: contentRow.episode,
   752→            contentTitle: contentRow.content_title || '',
   753→            contentPoster: contentRow.content_poster,
   754→          })
   755→          // After setPartyContent, re-check is_playing from the same query
   756→          if (contentRow.is_playing) {
   757→            useWatchPartyStore.getState().setPlaybackState({ isPlaying: true, currentTime: contentRow.playback_time || 0 })
   758→          }
   759→        } else if (!contentRow.content_id && current.currentParty.contentId) {
   760→          // Host cleared content — unlikely but handle it
   761→        }
   762→
   763→        // ── Sync party status (waiting → playing) ──────────────────
   764→        // CRITICAL FIX: Previously the content-heal interval only synced
   765→        // content_id. It did NOT sync the party status or is_playing flag.
   766→        // This meant that when the host clicked "Start" (changing status
   767→        // from 'waiting' to 'playing'), the member's store stayed at
   768→        // status='waiting' even though the DB had status='playing'.
   769→        // Without status='playing', the auto-play effect never triggers
   770→        // and the member's video never starts.
   771→        const dbStatus = contentRow.status as 'waiting' | 'playing' | 'ended'
   772→        if (dbStatus && dbStatus !== current.currentParty.status) {
   773→          console.log('[WatchParty] Content-heal: status changed from', current.currentParty.status, 'to', dbStatus)
   774→          current.setPartyStatus(dbStatus)
   775→          // When party transitions to 'playing', set partyStartTime so
   776→          // the auto-play effect triggers correctly for the member.
   777→          if (dbStatus === 'playing') {
   778→            useWatchPartyStore.getState().setPartyStartTime(contentRow.playback_time || 0)
   779→          }
   780→        }
   781→
   782→        // Sync is_playing even when content hasn't changed.
   783→        // The host may have started the party (is_playing: true) without
   784→        // changing content. Without this sync, members stuck at
   785→        // isPlaying=false would never start playing.
   786→        const dbIsPlaying = contentRow.is_playing
   787→        const dbPlaybackTime = contentRow.playback_time || 0
   788→        if (dbIsPlaying !== current.currentParty.playbackState.isPlaying) {
   789→          console.log('[WatchParty] Content-heal: isPlaying changed from', current.currentParty.playbackState.isPlaying, 'to', dbIsPlaying)
   790→          current.setPlaybackState({
   791→            isPlaying: dbIsPlaying,
   792→            currentTime: dbPlaybackTime,
   793→          })
   794→        }
   795→
   796→        // Sync member statuses from DB.
   797→        try {
   798→          const { data: memberRows } = await supabase
   799→            .from('watch_party_members')
   800→            .select('user_id, status')
   801→            .eq('party_id', _currentPartyId)
   802→            .in('status', ['joined', 'invited'])
   803→
   804→          if (!memberRows || memberRows.length === 0) return
   805→          const dbStatusMap = new Map(memberRows.map((m) => [m.user_id, m.status]))
   806→          const localMembers = current.currentParty.members
   807→          let needsUpdate = false
   808→          const updatedMembers = localMembers.map((m) => {
   809→            const dbMemberStatus = dbStatusMap.get(m.userId)
   810→            if (!dbMemberStatus) return m
   811→            if (m.memberStatus === 'invited' && dbMemberStatus === 'joined') {
   812→              needsUpdate = true
   813→              return { ...m, memberStatus: 'joined' as const }
   814→            }
   815→            return m
   816→          })
   817→          const localIds = new Set(localMembers.map((m) => m.userId))
   818→          const newDbMembers = memberRows.filter((m) => !localIds.has(m.user_id) && m.status === 'joined')
   819→          if (newDbMembers.length > 0) {
   820→            const newIds = newDbMembers.map((m) => m.user_id)
   821→            const { data: profiles } = await supabase
   822→              .from('profiles')
   823→              .select('id, display_name, avatar_url')
   824→              .in('id', newIds)
   825→            const profileMap = new Map((profiles || []).map((p) => [p.id, p]))
   826→            for (const row of newDbMembers) {
   827→              const profile = profileMap.get(row.user_id)
   828→              updatedMembers.push({
   829→                userId: row.user_id,
   830→                displayName: profile?.display_name || 'Unknown',
   831→                avatarUrl: profile?.avatar_url || null,
   832→                isHost: row.user_id === current.currentParty.hostId,
   833→                isTalking: false,
   834→                memberStatus: 'joined',
   835→              })
   836→              needsUpdate = true
   837→            }
   838→          }
   839→          const dbIds = new Set(memberRows.map((m) => m.user_id))
   840→          const removedMembers = updatedMembers.filter((m) => !dbIds.has(m.userId))
   841→          if (removedMembers.length > 0) {
   842→            needsUpdate = true
   843→          }
   844→          if (needsUpdate) {
   845→            current.setMembers(updatedMembers.filter((m) => dbIds.has(m.userId)))
   846→          }
   847→        } catch { /* non-critical */ }
   848→      } catch { /* non-critical */ } finally {
   849→        _syncInProgress = false
   850→      }
   851→    }, 3_000)
   852→
   853→    // ── Immediate DB refresh (not gated on WebSocket subscription) ──
   854→    // Fetch the latest party state right away so members see content
   855→    // and status changes that happened before they subscribed.
   856→    // This runs even if the WebSocket never connects.
   857→    fetchPartyFromDb(partyId).then((freshParty) => {
   858→      if (!freshParty) return
   859→      const currentStore = useWatchPartyStore.getState()
   860→      if (!currentStore.currentParty) return
   861→
   862→      const needsContentUpdate = freshParty.contentId &&
   863→        (freshParty.contentId !== currentStore.currentParty.contentId ||
   864→          freshParty.contentTitle !== currentStore.currentParty.contentTitle)
   865→      const needsStatusUpdate = freshParty.status !== currentStore.currentParty.status
   866→      const needsPlaybackUpdate = freshParty.playbackState.isPlaying !== currentStore.currentParty.playbackState.isPlaying
   867→
   868→      if (needsContentUpdate) {
   869→        currentStore.setPartyContent({
   870→          contentId: freshParty.contentId!,
   871→          mediaType: freshParty.mediaType!,
   872→          season: freshParty.season,
   873→          episode: freshParty.episode,
   874→          contentTitle: freshParty.contentTitle!,
   875→          contentPoster: freshParty.contentPoster,
   876→        })
   877→        if (freshParty.playbackState.isPlaying) {
   878→          useWatchPartyStore.getState().setPlaybackState({ isPlaying: true, currentTime: freshParty.playbackState.currentTime })
   879→        }
   880→      }
   881→      if (needsStatusUpdate) {
   882→        currentStore.setPartyStatus(freshParty.status)
   883→      }
   884→      if (needsPlaybackUpdate && !needsContentUpdate) {
   885→        currentStore.setPlaybackState({
   886→          isPlaying: freshParty.playbackState.isPlaying,
   887→          currentTime: freshParty.playbackState.currentTime,
   888→          pausedBy: freshParty.playbackState.pausedBy,
   889→        })
   890→      }
   891→    }).catch(() => { /* non-critical */ })
   892→
   893→    // ── Eagerly create voice manager BEFORE subscription ─────
   894→    // Previously the voice manager was only created inside the
   895→    // subscribe() callback when status === 'SUBSCRIBED'. If the
   896→    // Supabase Realtime WebSocket is stuck in CONNECTING (which
   897→    // happens when it falls back to REST for sends), the callback
   898→    // never fires and _voiceManager stays null forever — causing
   899→    // "Voice manager not available" on every PTT press.
   900→    // Creating it eagerly ensures it's available immediately.
   901→    //
   902→    // On iOS, we do NOT create a WebRTC voice manager — iOS uses
   903→    // voice clip recording instead. PeerConnections on iOS keep
   904→    // AVAudioSession in .playAndRecord mode, permanently ducking
   905→    // movie audio even after track.stop().
   906→    if (!isIOSDevice() && !_voiceManager) {
   907→      _voiceManager = new WebRtcVoiceManager(userId)
   908→      console.log('[WatchParty] WebRTC voice manager created eagerly for user', userId)
   909→
   910→      // Wire signal sender to Supabase broadcast (captures _partyChannel by reference)
   911→      _voiceManager.setSignalSender((targetUserId, signal) => {
   912→        if (!_partyChannel) return
   913→        let payload: WpWebrtcEvent
   914→        switch (signal.type) {
   915→          case 'offer':
   916→            payload = { t: 'webrtc-offer', targetUserId, fromUserId: userId, sdp: signal.sdp! }
   917→            break
   918→          case 'answer':
   919→            payload = { t: 'webrtc-answer', targetUserId, fromUserId: userId, sdp: signal.sdp! }
   920→            break
   921→          case 'ice-candidate':
   922→            payload = { t: 'webrtc-ice', targetUserId, fromUserId: userId, candidate: signal.candidate! }
   923→            break
   924→        }
   925→        wpBroadcast(_partyChannel, payload, `webrtc-${payload.t}`)
   926→      })
   927→
   928→      // Auto-play remote audio (manager handles HTMLAudioElement internally)
   929→      _voiceManager.setOnRemoteStream((remoteUserId, _stream) => {
   930→        console.log('[WatchParty] Remote stream received from', remoteUserId)
   931→        _voiceManager?.retryPausedAudio()
   932→      })
   933→
   934→      _voiceManager.setOnRemoteStreamRemoved((remoteUserId) => {
   935→        useWatchPartyStore.getState().setMemberTalking(remoteUserId, false)
   936→      })
   937→
   938→      // Mic init is DEFERRED to first PTT press (sendPttStart) to avoid
   939→      // altering the OS audio session before the iframe video player loads.
   940→      // On iOS/Android, getUserMedia switches AVAudioSession from Playback
   941→      // to PlayAndRecord which suppresses iframe audio output entirely.
   942→      // On desktop, it may lower output volume to prevent acoustic feedback.
   943→      // The voice manager OBJECT is still created eagerly here so that
   944→      // incoming WebRTC signals can be handled immediately.
   945→    }
   946→
   947→    // ── Postgres Changes listener (reliable real-time DB sync) ──────
   948→    // Supabase Broadcast is fire-and-forget — if the member's WebSocket
   949→    // isn't connected when the host picks content, the broadcast is lost.
   950→    // The content-heal interval (3s) is the safety net but too slow.
   951→    // Postgres Changes are triggered by actual DB writes (from the API),
   952→    // not client-side broadcasts, making them significantly more reliable.
   953→    // When the host's pickContent API call updates the DB, this listener
   954→    // fires and the member receives it near-instantly.
   955→    _partyChannel.on('postgres_changes',
   956→      {
   957→        event: 'UPDATE',
   958→        schema: 'public',
   959→        table: 'watch_parties',
   960→        filter: `id=eq.${partyId}`,
   961→      },
   962→      (payload: { new: Record<string, unknown>; old: Record<string, unknown> }) => {
   963→        const newRow = payload.new
   964→
   965→        // ── Content change ──────────────────────────────────
   966→        // Use FRESH store state for comparison (broadcast may have
   967→        // arrived between the Postgres Change event and this handler).
   968→        const freshStore = useWatchPartyStore.getState()
   969→        if (!freshStore.currentParty) return
   970→        const dbContentId = newRow.content_id as string | null
   971→        if (dbContentId && dbContentId !== freshStore.currentParty.contentId) {
   972→          console.log('[WatchParty] Postgres Change: content updated —', dbContentId)
   973→          freshStore.setPartyContent({
   974→            contentId: dbContentId,
   975→            mediaType: (newRow.media_type as 'movie' | 'tv') ?? 'movie',
   976→            season: (newRow.season as number) ?? null,
   977→            episode: (newRow.episode as number) ?? null,
   978→            contentTitle: (newRow.content_title as string) || '',
   979→            contentPoster: (newRow.content_poster as string) ?? null,
   980→          })
   981→          // Show toast only if the broadcast handler didn't already show
   982→          // one recently (dedup within CONTENT_PICK_TOAST_DEDUP_MS).
   983→          const now = Date.now()
   984→          const contentTitle = (newRow.content_title as string) || ''
   985→          if (now - _lastContentPickToastAt > CONTENT_PICK_TOAST_DEDUP_MS && contentTitle) {
   986→            _lastContentPickToastAt = now
   987→            toast.success(`Now watching: ${contentTitle}`)
   988→          }
   989→        }
   990→
   991→        // ── Status change (waiting → playing → ended) ───────
   992→        // Re-fetch store in case setPartyContent above mutated it
   993→        const statusStore = useWatchPartyStore.getState()
   994→        if (!statusStore.currentParty) return
   995→        const dbStatus = newRow.status as 'waiting' | 'playing' | 'ended'
   996→        if (dbStatus && dbStatus !== statusStore.currentParty.status) {
   997→          console.log('[WatchParty] Postgres Change: status changed from', statusStore.currentParty.status, 'to', dbStatus)
   998→          statusStore.setPartyStatus(dbStatus)
   999→          if (dbStatus === 'playing') {
  1000→            const dbPlaybackTime = (newRow.playback_time as number) || 0
  1001→            useWatchPartyStore.getState().setPlaybackState({ isPlaying: true, currentTime: dbPlaybackTime })
  1002→            useWatchPartyStore.getState().setPartyStartTime(dbPlaybackTime)
  1003→          }
  1004→          if (dbStatus === 'ended') {
  1005→            toast.info('Watch party has ended')
  1006→            useWatchPartyStore.getState().leaveParty()
  1007→            unsubscribePartyChannel()
  1008→          }
  1009→        }
  1010→
  1011→        // ── is_playing / playback_time sync ──────────────────
  1012→        const playStore = useWatchPartyStore.getState()
  1013→        if (!playStore.currentParty) return
  1014→        const dbIsPlaying = newRow.is_playing as boolean
  1015→        const dbPlaybackTime = (newRow.playback_time as number) || 0
  1016→        if (typeof dbIsPlaying === 'boolean' && dbIsPlaying !== playStore.currentParty.playbackState.isPlaying) {
  1017→          console.log('[WatchParty] Postgres Change: isPlaying changed from', playStore.currentParty.playbackState.isPlaying, 'to', dbIsPlaying)
  1018→          playStore.setPlaybackState({ isPlaying: dbIsPlaying, currentTime: dbPlaybackTime })
  1019→        }
  1020→      },
  1021→    )
  1022→
  1023→    // ── Broadcast events ────────────────────────────────────
  1024→    _partyChannel.on('broadcast', { event: 'wp' }, (payload: { payload: WpBroadcastEvent }) => {
  1025→      const event = payload.payload
  1026→      const store = useWatchPartyStore.getState()
  1027→
  1028→      switch (event.t) {
  1029→        case 'member-joined':
  1030→          if (store.currentParty) {
  1031→            const existing = store.currentParty.members.find((m) => m.userId === event.member.userId)
  1032→            if (existing) {
  1033→              // Update existing member's status from 'invited' → 'joined'
  1034→              store.setMembers(
  1035→                store.currentParty.members.map((m) =>
  1036→                  m.userId === event.member.userId ? { ...m, ...event.member, memberStatus: event.member.memberStatus || 'joined' } : m
  1037→                )
  1038→              )
  1039→              console.log(`[WatchParty] Member ${event.member.displayName} status updated to '${event.member.memberStatus || 'joined'}'`)
  1040→            } else {
  1041→              store.setMembers([...store.currentParty.members, event.member])
  1042→              console.log(`[WatchParty] New member ${event.member.displayName} added with status '${event.member.memberStatus || 'joined'}'`)
  1043→            }
  1044→            toast.success(`${event.member.displayName} joined the watch party`)
  1045→
  1046→            // NOTE: We intentionally do NOT refresh members from DB here.
  1047→            // The broadcast carries the authoritative member status ('joined').
  1048→            // A DB refresh at this point races with the accept API write
  1049→            // and can overwrite the correct 'joined' status with stale 'invited'
  1050→            // data from a read that occurred before the DB transaction committed.
  1051→            // The 30-second periodic sync handles any missed updates.
  1052→
  1053→            // Re-broadcast current content to the new member.
  1054→            // If the host picked content before the member subscribed,
  1055→            // the member missed the content-picked broadcast and their
  1056→            // panel would stay on "Waiting for host to pick content...".
  1057→            // The content-picked handler on the member side only shows
  1058→            // a toast if the content actually changed, so existing
  1059→            // members won't see duplicate toasts.
  1060→            if (event.member.userId !== userId && store.currentParty.contentId && _partyChannel) {
  1061→              wpBroadcast(_partyChannel, {
  1062→                  t: 'content-picked',
  1063→                  contentId: store.currentParty.contentId,
  1064→                  mediaType: store.currentParty.mediaType!,
  1065→                  season: store.currentParty.season,
  1066→                  episode: store.currentParty.episode,
  1067→                  contentTitle: store.currentParty.contentTitle!,
  1068→                  contentPoster: store.currentParty.contentPoster,
  1069→                } as WpContentPickedEvent, 'content-rebroadcast')
  1070→              console.log(`[WatchParty] Re-broadcast content-picked to new member ${event.member.displayName}`)
  1071→            }
  1072→

  1247→            clearTalkingTimeout(event.userId)

  1249→          }
  1250→          break

  1254→          // Update the member entry in the current party so the UI
  1255→          // reflects the change instantly in the members list.

  1258→              displayName: event.displayName,
  1259→              avatarUrl: event.avatarUrl,
  1260→            })
  1261→          }
  1262→          break
  1263→

  1267→          if (event.endedBy && store.currentParty && event.endedBy !== store.currentParty.hostId) {

  1275→          unsubscribePartyChannel()

  1281→        // all WebRTC signaling to avoid triggering AVAudioSession changes.

  1284→            _voiceManager?.handleOffer(event.fromUserId, event.sdp)

  1286→          break
  1287→
  1288→        case 'webrtc-answer':
  1289→          if (!isIOSDevice() && event.targetUserId === userId && _voiceReady) {
  1290→            _voiceManager?.handleAnswer(event.fromUserId, event.sdp)
  1291→          }
  1292→          break
  1293→
  1294→        case 'webrtc-ice':
  1295→          if (!isIOSDevice() && event.targetUserId === userId && _voiceReady) {
  1296→            _voiceManager?.handleIceCandidate(event.fromUserId, event.candidate)
  1297→          }
  1298→          break
  1299→

  1301→        case 'voice-clip': {
  1302→          const clipEvent = event as unknown as WpVoiceClipEvent
  1303→          if (clipEvent.fromUserId && clipEvent.fromUserId !== userId) {
  1304→            // Show speaking indicator for the duration of the clip
  1305→            store.setMemberTalking(clipEvent.fromUserId, true)
  1306→            playVoiceClip(
  1307→              clipEvent.fromUserId,
  1308→              clipEvent.audio,
  1309→              clipEvent.mimeType,
  1310→              () => {
  1311→                // Clear speaking indicator when clip finishes playing
  1312→                store.setMemberTalking(clipEvent.fromUserId, false)
  1313→              },
  1314→            )
  1315→          }
  1316→          break
  1317→        }

  1321→    // ── Presence events ─────────────────────────────────────
  1322→    _partyChannel.on('presence', { event: 'sync' }, () => {
  1323→      if (!_partyChannel) return
  1324→      const store = useWatchPartyStore.getState()
  1325→      const newState = _partyChannel.presenceState()
  1326→
  1327→      // Build talking map from presence, but EXCLUDE the local user.
  1328→      // The local user's talking state is managed by isPttActive,
  1329→      // not talkingMembers. Including it here causes the wrong
  1330→      // member to show as speaking when the local user presses PTT.
  1331→      const talkingMap = new Map<string, boolean>()
  1332→      Object.values(newState).forEach((presences) => {
  1333→        presences.forEach((p) => {
  1334→          const pData = p as unknown as { userId: string; isTalking: boolean }
  1335→          if (pData.isTalking && pData.userId !== userId) {
  1336→            talkingMap.set(pData.userId, true)
  1337→          }
  1338→        })
  1339→      })
  1340→
  1341→      const currentTalking = store.talkingMembers
  1342→      talkingMap.forEach((_, uid) => {
  1343→        // Skip if this user explicitly stopped PTT — the presence data
  1344→        // is stale (isTalking: true) but the ptt-stop broadcast already
  1345→        // cleared the indicator. Only a new ptt-start broadcast can
  1346→        // re-activate the talking indicator for this user.
  1347→        if (_pttStoppedUsers.has(uid)) return
  1348→        if (!currentTalking.has(uid)) store.setMemberTalking(uid, true)
  1349→      })
  1350→      currentTalking.forEach((_, uid) => {
  1351→        if (!talkingMap.has(uid) && uid !== userId) store.setMemberTalking(uid, false)
  1352→      })
  1353→    })
  1354→
  1355→    _partyChannel.on('presence', { event: 'leave' }, (payload: { leftPresences: { userId: string }[] }) => {
  1356→      const store = useWatchPartyStore.getState()
  1357→      const myUserId = useAuthStore.getState().user?.id
  1358→
  1359→      payload.leftPresences.forEach((p) => {
  1360→        store.setMemberTalking(p.userId, false)
  1361→        _pttStoppedUsers.delete(p.userId)
  1362→

  1364→          store.setPlaybackState({ pausedBy: null })
  1365→          store.clearPauseNotification()

  1367→        }
  1368→
  1369→        // If the HOST left presence, start a 30-second timer.
  1370→        // If the host doesn't rejoin within 30s, the party auto-ends.
  1371→        // This handles: host closed app, killed process, lost connection
  1372→        // for extended period. It does NOT fire on brief network blips
  1373→        // because the host's presence will rejoin within a few seconds.
  1374→        if (
  1375→          p.userId !== myUserId &&
  1376→          store.currentParty?.hostId === p.userId &&
  1377→          !store.isHost
  1378→        ) {

  1380→          _hostLeftAt = Date.now()

  1383→            const current = useWatchPartyStore.getState()
  1384→            if (!current.currentParty || !current.isInParty) return
  1385→            // Double-check: host presence might have rejoined
  1386→            const presence = _partyChannel?.presenceState()
  1387→            const hostPresent = presence && Object.values(presence).some((presences) =>
  1388→              presences.some((pr) => (pr as unknown as { userId: string }).userId === p.userId)
  1389→            )
  1390→            if (hostPresent) {
  1391→              _hostLeftAt = null
  1392→              return
  1393→            }
  1394→            console.log('[WatchParty] Host absent for 1s — auto-ending party for members')
  1395→            toast.info('Host disconnected — watch party ended')
  1396→            current.leaveParty()
  1397→            unsubscribePartyChannel()
  1398→            _hostLeftAt = null
  1399→          }, HOST_ABSENCE_TIMEOUT_MS)
  1400→        }
  1401→
  1402→        // If a MEMBER left presence and we're the host, start a timer
  1403→        // to remove them from the party after 30 seconds of absence.
  1404→        // This handles: member closed app, killed process, lost connection.
  1405→        // Presence already has ~30s built-in tolerance from Supabase heartbeat,
  1406→        // so by the time this event fires, the member has been gone ~60s total.
  1407→        if (
  1408→          p.userId !== myUserId &&
  1409→          store.isHost &&
  1410→          store.currentParty?.hostId === myUserId &&
  1411→          p.userId !== store.currentParty?.hostId
  1412→        ) {

  1414→          const existingTimer = _memberLeftTimers.get(p.userId)
  1415→          if (existingTimer) clearTimeout(existingTimer)
  1416→

  1418→          _memberLeftTimers.set(absentUserId, setTimeout(async () => {
  1419→            const current = useWatchPartyStore.getState()
  1420→            if (!current.currentParty || !current.isInParty || !current.isHost) return
  1421→            // Double-check: member presence might have rejoined
  1422→            const presence = _partyChannel?.presenceState()
  1423→            const memberPresent = presence && Object.values(presence).some((presences) =>
  1424→              presences.some((pr) => (pr as unknown as { userId: string }).userId === absentUserId)
  1425→            )
  1426→            if (memberPresent) {
  1427→              _memberLeftTimers.delete(absentUserId)
  1428→              return
  1429→            }
  1430→            // Member absent for 30s after presence leave — remove from party
  1431→            console.log('[WatchParty] Member', absentUserId, 'absent for 30s — removing from party')
  1432→            const partyId = _currentPartyId
  1433→            // Remove from local member list
  1434→            current.setMembers(current.currentParty.members.filter((m) => m.userId !== absentUserId))
  1435→            // Clean up WebRTC peer connection
  1436→            _voiceManager?.removePeer(absentUserId)

  1438→            const absentMember = current.currentParty.members.find((m) => m.userId === absentUserId)

  1440→              wpBroadcast(_partyChannel, {
  1441→                t: 'member-left',
  1442→                userId: absentUserId,
  1443→                displayName: absentMember.displayName,
  1444→                leftBy: useAuthStore.getState().user?.id ?? absentUserId, // host is removing the member

  1447→            // Update DB: set member status to 'left'
  1448→            if (partyId) {
  1449→              wpApi('remove-member', { partyId, targetUserId: absentUserId }).catch(() => { /* non-critical */ })

  1533→                  updatedStore.setPlaybackState({ isPlaying: true, currentTime: freshParty.playbackState.currentTime })

  1538→              }

  1541→                  isPlaying: freshParty.playbackState.isPlaying,
  1542→                  currentTime: freshParty.playbackState.currentTime,
  1543→                  pausedBy: freshParty.playbackState.pausedBy,

  1550→              // the correct data, but this DB fetch can race with the write
  1551→              // on PWA (Supabase read replicas) and return stale 'invited'.
