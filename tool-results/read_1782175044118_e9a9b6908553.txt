   260→    async function init() {

   266→          // ── Null session: NEVER clear state from this handler ──
   267→          // signOut() fires SIGNED_OUT asynchronously (queued by GoTrue).

   269→          // a new signInWithPassword() + setStatus('authenticated'), wiping

   282→          // login handler sets auth state directly via setAuth + setStatus.

   284→

   286→          if (!navigator.onLine) return;

   289→            // Session refreshed or user updated — re-sync auth + profile
   290→            setAuth(session.user, session);
   291→            const profile = await getMyProfile();

   293→          } catch {
   294→            // Non-critical: transient error (network blip, Supabase hiccup).
   295→            // The next heartbeat or online event will retry profile sync.
   296→          }
   297→        },
   298→      ).data.subscription;

   302→        const { data: { session } } = await supabase.auth.getSession();

   304→        if (!session) {
   305→          // OFFLINE RECOVERY: getSession() may return null when Supabase cleared
   306→          // the stored session during a failed token refresh. If we're offline
   307→          // and have a cached session in localStorage, recover from it instead
   308→          // of wiping the auth state.
   309→          if (!navigator.onLine) {
   310→            try {
   311→              const raw = localStorage.getItem('streamvault-auth-token');
   312→              if (raw) {
   313→                const cached = JSON.parse(raw);
   314→                if (cached?.access_token) {
   315→                  // Keep the authenticated state from bootstrapFromCache
   316→                  return;
   317→                }
   318→              }
   319→            } catch { /* ignore parse errors */ }
   320→          }
   321→          if (mounted) {
   322→            setProfile(null);
   323→            setStatus('unauthenticated');
   324→          }
   325→          return;
   326→        }
   327→
   328→        if (mounted) setAuth(session.user, session);
   329→
   330→        const metaCompleted = !!session.user.user_metadata?.profile_completed;
   331→        const profile = await getMyProfile();
   332→        if (!mounted) return;
   333→
   334→        setProfile(profile);
   335→
   336→        if (metaCompleted || (profile && profile.display_name.trim().length > 0)) {
   337→          setStatus('authenticated');
   338→          if (session.access_token) {
   339→            fetchWatchlistFromServer(session.access_token);
   340→            // Bump updated_at to signal user activity (throttled to 1h)
   341→            touchProfile(profile?.updated_at);
   342→          }
   343→        } else {
   344→          setStatus('needs_profile');
   345→        }
   346→      } catch {
   347→        if (mounted) {
   348→          const current = useAuthStore.getState().status;
   349→          // OFFLINE RECOVERY: bootstrapFromCache() already set status to
   350→          // 'authenticated'. Don't overwrite it with 'unauthenticated' just
   351→          // because getMyProfile() failed (expected when offline).
   352→          if (current === 'loading') {

   355→              setProfile(cached);
   356→              setStatus('authenticated');
   357→              // Only mark offline if actually offline — getMyProfile() can
   358→              // fail for non-offline reasons (RLS, Supabase hiccup, etc.)
   359→              if (!navigator.onLine) setOffline(true);
   360→            } else {
   361→              setStatus('unauthenticated');
   362→            }
   363→          } else if (current === 'authenticated') {
   364→            // Already recovered by bootstrap — only mark offline if actually offline
   365→            if (!navigator.onLine) setOffline(true);
   366→          }
   367→        }
   368→      }
   369→    }
   370→
   371→    init();
   372→
   373→    return () => {
   374→      mounted = false;
   375→      if (subscription) subscription.unsubscribe();
   376→    };
